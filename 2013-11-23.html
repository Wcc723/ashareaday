透過kinetic可以輕鬆完成許多canvas的事件，而功能不斷地加入，也讓這小東西越來越有趣，今天再新增文字欄位，和上次不同的事，今天的文字可以一直"新增"且可以"拖曳"！

Demo

現在可以先把圖放入，接下來在輸入文字。

請拖曳圖片到此範圍內(別用太大的圖片~ 我沒限制嘿)

<input id="text" type="text" placeholder="請輸入些文字後按Enter">

var stage = new Kinetic.Stage({
      container: 'canvas',
      width: 600,
      height: 300
    });


var textlayer = new Kinetic.Layer();
  //新增一個文字圖層
  $('#text').on('change',function(){
    var textContent = $(this).val();
    //把input欄位內的文字取出來
    var simpleText = new Kinetic.Text({
      //Kinetic所提供的文字物件
        x: stage.getWidth() / 2,        
        y: 15,
        //x,y位置
        text: textContent,
        //文字內容
        fontSize: 40,
        //文字大小
        fontFamily: 'Arial',
        //字型
        fill: '#fff',
        //文字顏色
        stroke: '#E80C7A',
        //筆畫
        draggable: true
        //開啟拖曳
      });
    textlayer.add(simpleText);
    //把文字物件加進去圖層
    stage.add(textlayer);
    //把圖層加入場景
  });
